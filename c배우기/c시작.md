```c
#include <stdio.h>

int main(void) {
    printf("여다영바보 다영%.2lf\n",9.9);
    printf("여다영%d살 똥 좋아함\n",12);
    printf("hoho\bb\tgaga\n");
    printf("hoho\rgaga\a");
    
    return 0;
}
```
